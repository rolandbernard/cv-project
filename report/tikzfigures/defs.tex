\newcommand{\drawbox}[2]{
  \pgfmathsetmacro{\W}{#1}
  \pgfmathsetmacro{\H}{#2}

  \coordinate (A) at (0,0);
  \coordinate (B) at (0,\H);
  \coordinate (C) at (\W,\H);
  \coordinate (D) at (\W,0);
  
  \draw[thick] 
    (A) -- (B) -- (C) -- (D) -- (A);
}

\newcommand{\drawfull}[3]{
  \pgfmathsetmacro{\W}{#1}
  \pgfmathsetmacro{\H}{#2}
  \pgfmathsetmacro{\h}{#3}

  \coordinate (A) at (0,0);
  \coordinate (B) at (0,\H);
  \coordinate (C) at (\W,\H-\h);
  \coordinate (D) at (\W,\h);
  
  \draw[dashed, thick] 
    (A) -- (B) -- (C) -- (D) -- (A);
}

\newcommand{\drawcuboid}[3]{
  \pgfmathsetmacro{\L}{#1}
  \pgfmathsetmacro{\W}{#2}
  \pgfmathsetmacro{\H}{#3}

  \coordinate (O) at (0,0,0);
  \coordinate (A) at (\L,0,0);
  \coordinate (B) at (\L,\W,0);
  \coordinate (C) at (0,\W,0);
  \coordinate (D) at (0,0,\H);
  \coordinate (E) at (\L,0,\H);
  \coordinate (F) at (\L,\W,\H);
  \coordinate (G) at (0,\W,\H);
  
  \draw
    (O) -- (A)
    (O) -- (C)
    (O) -- (D);
  
  \draw[thick] 
    (A) -- (B) -- (C) -- (G) -- (F) -- (B)
    (A) -- (E) -- (F)
    (C) -- (G)
    (D) -- (E)
    (D) -- (G);
}

\newcommand{\drawconv}[5]{
  \pgfmathsetmacro{\L}{#1}
  \pgfmathsetmacro{\W}{#2}
  \pgfmathsetmacro{\H}{#3}
  \pgfmathsetmacro{\w}{#4}
  \pgfmathsetmacro{\h}{#5}

  \coordinate (O) at (0,0,0);
  \coordinate (A) at (\L,\w,\h);
  \coordinate (B) at (\L,\W-\w,\h);
  \coordinate (C) at (0,\W,0);
  \coordinate (D) at (0,0,\H);
  \coordinate (E) at (\L,\w,\H-\h);
  \coordinate (F) at (\L,\W-\w,\H-\h);
  \coordinate (G) at (0,\W,\H);
  
  \draw[dashed]
    (O) -- (A)
    (O) -- (C)
    (O) -- (D);
  
  \draw[dashed, thick] 
    (A) -- (B) -- (C) -- (G) -- (F) -- (B)
    (A) -- (E) -- (F)
    (C) -- (G)
    (D) -- (E)
    (D) -- (G);
}
